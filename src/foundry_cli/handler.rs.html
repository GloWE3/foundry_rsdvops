4/// A custom context type for Foundry specific error reporting via `eyre`
5#[derive(Debug)]
6pub struct Handler;
7
8impl EyreHandler for Handler {
9    fn debug(&self, error: &(dyn Error + 'static), f: &mut fmt::Formatter<'_>) -> fmt::Result {
10        if f.alternate() {
11            return fmt::Debug::fmt(error, f)
12        }
13        let errors = foundry_common::errors::dedup_chain(error);
14
15        let (error, sources) = errors.split_first().unwrap();
16        write!(f, "{error}")?;
17
18        if !sources.is_empty() {
19            write!(f, "\n\nContext:")?;
20
21            let multiple = sources.len() > 1;
22            for (n, error) in sources.iter().enumerate() {
23                writeln!(f)?;
24                if multiple {
25                    write!(f, "- Error #{n}: {error}")?;
26                } else {
27                    write!(f, "- {error}")?;
28                }
29            }
30        }
31
32        Ok(())
33    }
34}

36/// Installs the Foundry [eyre] and [panic](mod@std::panic) hooks as the global ones.
37///
38/// # Details
39///
40/// By default a simple user-centric handler is installed, unless
41/// `FOUNDRY_DEBUG` is set in the environment, in which case a more
42/// verbose debug-centric handler is installed.
43///
44/// Panics are always caught by the more debug-centric handler.
45pub fn install() {
46    if std::env::var_os("RUST_BACKTRACE").is_none() {
47        std::env::set_var("RUST_BACKTRACE", "1");
48    }
49
50    let panic_section =
51        "This is a bug. Consider reporting it at https://github.com/foundry-rs/foundry";
52    let (panic_hook, debug_eyre_hook) =
53        color_eyre::config::HookBuilder::default().panic_section(panic_section).into_hooks();
54    panic_hook.install();
55    let eyre_install_result = if std::env::var_os("FOUNDRY_DEBUG").is_some() {
56        debug_eyre_hook.install()
57    } else {
58        eyre::set_hook(Box::new(|_| Box::new(Handler)))
59    };
60    if let Err(e) = eyre_install_result {
61        debug!("failed to install eyre error hook: {e}");
62    }
63}
