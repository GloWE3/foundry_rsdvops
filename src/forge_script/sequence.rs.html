14pub enum ScriptSequenceKind {
15    Single(ScriptSequence),
16    Multi(MultiChainSequence),
17}
18
19impl ScriptSequenceKind {
20    pub fn save(&mut self, silent: bool, save_ts: bool) -> Result<()> {
21        match self {
22            Self::Single(sequence) => sequence.save(silent, save_ts),
23            Self::Multi(sequence) => sequence.save(silent, save_ts),
24        }
25    }
26
27    pub fn sequences(&self) -> &[ScriptSequence] {
28        match self {
29            Self::Single(sequence) => std::slice::from_ref(sequence),
30            Self::Multi(sequence) => &sequence.deployments,
31        }
32    }
33
34    pub fn sequences_mut(&mut self) -> &mut [ScriptSequence] {
35        match self {
36            Self::Single(sequence) => std::slice::from_mut(sequence),
37            Self::Multi(sequence) => &mut sequence.deployments,
38        }
39    }
40    /// Updates underlying sequence paths to not be under /dry-run directory.
41    pub fn update_paths_to_broadcasted(
42        &mut self,
43        config: &Config,
44        sig: &str,
45        target: &ArtifactId,
46    ) -> Result<()> {
47        match self {
48            Self::Single(sequence) => {
49                sequence.paths =
50                    Some(ScriptSequence::get_paths(config, sig, target, sequence.chain, false)?);
51            }
52            Self::Multi(sequence) => {
53                (sequence.path, sequence.sensitive_path) =
54                    MultiChainSequence::get_paths(config, sig, target, false)?;
55            }
56        };
57
58        Ok(())
59    }
60}
61
62impl Drop for ScriptSequenceKind {
63    fn drop(&mut self) {
64        if let Err(err) = self.save(false, true) {
65            error!(?err, "could not save deployment sequence");
66        }
67    }
